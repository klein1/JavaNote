void quickSort(int[] a, int i, int j){
    if(i < j){
        int left = i;
        int right = j;
        int pivot = a[i];
        while(left <= right){
            while(left <= right && a[left] <= pivot) left++;
            while(left <= right && a[right] > pivot) right--;
            if(left < right){
                int temp = a[left];
                a[left] = a[right];
                a[right] = temp;
            }
        }
        a[i] = a[right];
        a[right] = piv;
            
        quickSort(a, i, right-1);
        quickSort(a, right+1, j);
    }
}
